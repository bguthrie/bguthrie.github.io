{{/* OG Image Generator Produces a 1200x630 JPG with: - The post's header image (or fallback) as background - A gradient overlay darkening the bottom half - The blog avatar in the upper-left corner - The post title left-aligned at the bottom */}} {{- $font := resources.Get "fonts/Inter-Bold.ttf" -}} {{/* Get background image: page bundle image or global fallback */}} {{- $bg := false -}} {{- with .Params.image -}} {{- $bg = $.Resources.Get . -}} {{- end -}} {{- if not $bg -}} {{- $bg = resources.Get "images/og-fallback.webp" -}} {{- end -}} {{/* Load gradient overlay and avatar */}} {{- $gradient := resources.Get "images/og-gradient.png" -}} {{- $avatar := resources.Get "img/profile.png" -}} {{/* Word-wrap the title to ~30 chars per line */}} {{- $title := .Title -}} {{- $words := split $title " " -}} {{- $lines := slice -}} {{- $currentLine := "" -}} {{- range $words -}} {{- $candidate := . -}} {{- if $currentLine -}} {{- $candidate = printf "%s %s" $currentLine . -}} {{- end -}} {{- if and (gt (len $candidate) 30) $currentLine -}} {{- $lines = $lines | append $currentLine -}} {{- $currentLine = . -}} {{- else -}} {{- $currentLine = $candidate -}} {{- end -}} {{- end -}} {{- if $currentLine -}} {{- $lines = $lines | append $currentLine -}} {{- end -}} {{- $wrappedTitle := delimit $lines "\n" -}} {{/* Process the image: resize, gradient overlay, avatar, title text */}} {{- $img := $bg.Process "fill 1200x630 Center jpg" -}} {{/* Apply gradient overlay (dark at bottom, transparent at top) */}} {{- $img = $img | images.Filter (images.Overlay $gradient 0 0) -}} {{/* Add avatar in upper-left corner (scaled to 96x96) */}} {{- $avatarSmall := $avatar.Process "fit 96x96 png" -}} {{- $img = $img | images.Filter (images.Overlay $avatarSmall 40 40) -}} {{/* Add title text: left-aligned, near the bottom */}} {{- $img = $img | images.Filter (images.Text $wrappedTitle (dict "size" 52 "color" "#ffffff" "font" $font "alignx" "left" "aligny" "bottom" "x" 40 "y" 590 "linespacing" 10)) -}} {{- return $img -}}
